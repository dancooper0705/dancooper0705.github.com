## source
```python
from PIL import Image, ImageFilter

def main():
    img1 = Image.new("RGB", (16,16))
    for i in range(8):
        for j in range(8):-
            img1.putpixel((i,j), 0x0000ff)
    #img1.show()
    b1 = img1.tobytes()
    img2 = Image.new("RGB", (16,16))
    img2.frombytes(b1)
    #img2.show()
    b2 = img2.tobytes()
    if b1 == b2: 
        print("b1 == b2")
    else:
        print("b1 != b2")
    print("b1: " + str(b1))
    print("b2: " + str(b2))

if __name__ == "__main__":
    main()
```
